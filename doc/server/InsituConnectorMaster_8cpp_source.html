70 errorCode InsituConnectorMaster::run()
71 {
72  listen(sockfd,5);
73  struct sockaddr_in cli_addr;
74  socklen_t clilen = sizeof(cli_addr);
75 
76  struct pollfd fd_array[MAX_SOCKETS];
77  memset(fd_array,0,sizeof(fd_array));
78  std::vector< InsituConnectorContainer* > con_array = std::vector< InsituConnectorContainer* >(MAX_SOCKETS,NULL);
79 
80  fd_array[0].fd = sockfd;
81  fd_array[0].events = POLLIN;
82  int fdnum = 1;
83 
84  while (!force_exit)
85  {
86  int rv = poll(fd_array, fdnum, 1000); //1s timeout
87  if (rv < 0)
88  {
89  fprintf(stderr,"Error while calling poll\n");
90  return -1;
91  }
92  if (rv)
93  {
94  //First some extra sausage for the listening sockfd
95  if (fd_array[0].revents == POLLIN)
96  {
97  int newsockfd = 1;
98  while (newsockfd >= 0)
99  {
100  newsockfd = accept(sockfd, (struct sockaddr *) &cli_addr, &clilen);
101  if (newsockfd >= 0)
102  {
103  InsituConnector* insituConnector = new InsituConnector(newsockfd,nextFreeNumber++);
104  insituConnector->jlcb.count = 0;
105  InsituConnectorContainer* d = new InsituConnectorContainer();
106  d->connector = insituConnector;
107  con_array[fdnum] = d;
108  fd_array[fdnum].fd = newsockfd;
109  fd_array[fdnum].events = POLLIN;
110  insituConnectorList.push_back(d);
111  fdnum++;
112  }
113  }
114  }
115  for (int i = 1; i < fdnum; i++)
116  {
117  if (fd_array[i].revents & POLLIN)
118  {
119  while (1)
120  {
121  int add = recv(fd_array[i].fd,&(con_array[i]->connector->jlcb.buffer[con_array[i]->connector->jlcb.count]),4096,MSG_DONTWAIT);
122  if (add > 0)
123  con_array[i]->connector->jlcb.count += add;
124  else
125  break;
126  }
127  con_array[i]->connector->jlcb.pos = 0;
128  con_array[i]->connector->jlcb.buffer[con_array[i]->connector->jlcb.count] = 0;
129  bool closed = false;
130  if (con_array[i]->connector->jlcb.count > 0)
131  {
132  con_array[i]->connector->jlcb.buffer[con_array[i]->connector->jlcb.count] = 0;
133  json_error_t error;
134  int last_working_pos = 0;
135  while (json_t * content = json_load_callback(json_load_callback_function,&con_array[i]->connector->jlcb,JSON_DISABLE_EOF_CHECK,&error))
136  {
137  last_working_pos = con_array[i]->connector->jlcb.pos;
138  MessageContainer* message = new MessageContainer(NONE,content);
139  MessageType type = message->type;
140  if (type == REGISTER)
141  json_object_set_new( message->json_root, "id", json_integer( con_array[i]->connector->getID() ) );
142  long long uid = json_integer_value( json_object_get(content, "uid") );
143  con_array[i]->connector->clientSendMessage(message);
144  if (type == EXIT_PLUGIN)
145  {
146  closed = true;
147  break;
148  }
149  else
150  {
151  //send, which uid we just got
152  char buffer[32];
153  sprintf(buffer,"{\"done\": %lld}", uid );
154  int l = strlen(buffer);
155  if ( send(fd_array[i].fd,buffer,l,MSG_NOSIGNAL) < l )
156  {
157  MessageContainer* message = new MessageContainer(EXIT_PLUGIN,json_object());
158  json_object_set_new( message->json_root, "type", json_string( "exit" ) );
159  con_array[i]->connector->clientSendMessage(message);
160  closed = true;
161  break;
162  }
163  }
164  }
165  //If the whole json message was not received yet, we need to keep the start
166  if ( error.position != 1 || strcmp(error.text,"'[' or '{' expected near end of file") != 0 )
167  {
168  for (int j = 0; j < con_array[i]->connector->jlcb.count - last_working_pos; j++)
169  con_array[i]->connector->jlcb.buffer[j] = con_array[i]->connector->jlcb.buffer[j + last_working_pos];
170  con_array[i]->connector->jlcb.count -= last_working_pos;
171  }
172  else
173  con_array[i]->connector->jlcb.count = 0;
174  }
175  else //Closed
176  {
177  MessageContainer* message = new MessageContainer(EXIT_PLUGIN,json_object());
178  json_object_set_new( message->json_root, "type", json_string( "exit" ) );
179  con_array[i]->connector->clientSendMessage(message);
180  closed = true;
181  }
182  if (closed)
183  {
184  close(fd_array[i].fd);
185  fdnum--;
186  for (int j = i; j < fdnum; j++)
187  {
188  fd_array[j] = fd_array[j+1];
189  con_array[j] = con_array[j+1];
190  }
191  memset(&(fd_array[fdnum]),0,sizeof(fd_array[fdnum]));
192  }
193  }
194  }
195  }
196  }
197 }
