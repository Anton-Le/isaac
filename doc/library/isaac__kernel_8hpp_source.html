235  isaac_float result = isaac_float(0);
236

276  <

278  typename TSource,
279  typename TColor,
280  typename TPos,
281  typename TLocalSize,

290  ISAAC_HOST_DEVICE_INLINE void operator()(
291  const NR& nr,
292  const TSource& source,
293  TColor& color,
294  const TPos& pos,
295  const TLocalSize& local_size,
296  const TTransferArray& transferArray,
297  const TSourceWeight& sourceWeight,
298  const TPointerArray& pointerArray,
299  TFeedback& feedback,
300  const TStep& step,
301  const TStepLength& stepLength,
302  const TScale& scale
303  ) const
304  {
305  if ( mpl::at_c< TFilter, NR::value >::type::value )
306  {
307  isaac_float result = get_value< TInterpolation, NR >( source, pos, pointerArray, local_size, scale );
308  isaac_int lookup_value = isaac_int( round(result * isaac_float( Ttransfer_size ) ) );
309  if (lookup_value < 0 )
310  lookup_value = 0;
311  if (lookup_value >= Ttransfer_size )
312  lookup_value = Ttransfer_size - 1;
313  isaac_float4 value = transferArray.pointer[ NR::value ][ lookup_value ];
314  if (TIsoSurface)
315  {
316  if (value.w >= isaac_float(0.5))
317  {
318  isaac_float3 left = {-1, 0, 0};
319  left = left + pos;
320  if (!TSource::has_guard && TSource::persistent)
321  check_coord( left, local_size);
322  isaac_float3 right = { 1, 0, 0};
323  right = right + pos;
324  if (!TSource::has_guard && TSource::persistent)
325  check_coord( right, local_size );
326  isaac_float d1;
327  if (TInterpolation)
328  d1 = right.x - left.x;
329  else
330  d1 = isaac_int(right.x) - isaac_int(left.x);
331 
332  isaac_float3 up = { 0,-1, 0};
333  up = up + pos;
334  if (!TSource::has_guard && TSource::persistent)
335  check_coord( up, local_size );
336  isaac_float3 down = { 0, 1, 0};
337  down = down + pos;
338  if (!TSource::has_guard && TSource::persistent)
339  check_coord( down, local_size );
340  isaac_float d2;
341  if (TInterpolation)
342  d2 = down.y - up.y;
343  else
344  d2 = isaac_int(down.y) - isaac_int(up.y);
345 
346  isaac_float3 front = { 0, 0,-1};
347  front = front + pos;
348  if (!TSource::has_guard && TSource::persistent)
349  check_coord( front, local_size );
350  isaac_float3 back = { 0, 0, 1};
351  back = back + pos;
352  if (!TSource::has_guard && TSource::persistent)
353  check_coord( back, local_size );
354  isaac_float d3;
355  if (TInterpolation)
356  d3 = back.z - front.z;
357  else
358  d3 = isaac_int(back.z) - isaac_int(front.z);
359 
360  isaac_float3 gradient=
361  {
362  (get_value< TInterpolation, NR >( source, right, pointerArray, local_size, scale ) -
363  get_value< TInterpolation, NR >( source, left, pointerArray, local_size, scale )) / d1,
364  (get_value< TInterpolation, NR >( source, down, pointerArray, local_size, scale ) -
365  get_value< TInterpolation, NR >( source, up, pointerArray, local_size, scale )) / d2,
366  (get_value< TInterpolation, NR >( source, back, pointerArray, local_size, scale ) -
367  get_value< TInterpolation, NR >( source, front, pointerArray, local_size, scale )) / d3
368  };
369  isaac_float l = sqrt(
370  gradient.x * gradient.x +
371  gradient.y * gradient.y +
372  gradient.z * gradient.z
373  );
374  if (l == isaac_float(0))
375  color = value;
376  else
377  {
378  gradient = gradient / l;
379  isaac_float3 light = step / stepLength;
380  isaac_float ac = fabs(
381  gradient.x * light.x +
382  gradient.y * light.y +
383  gradient.z * light.z );
384  #if ISAAC_SPECULAR == 1
385  color.x = value.x * ac + ac * ac * ac * ac;
386  color.y = value.y * ac + ac * ac * ac * ac;
387  color.z = value.z * ac + ac * ac * ac * ac;
388  #else
389  color.x = value.x * ac;
390  color.y = value.y * ac;
391  color.z = value.z * ac;
392  #endif
393  }
394  color.w = isaac_float(1);
395  feedback = 1;
396  }
397  }
398  else
399  {
400  value.w *= sourceWeight.value[ NR::value ];
401  color.x = color.x + value.x * value.w;
402  color.y = color.y + value.y * value.w;
403  color.z = color.z + value.z * value.w;
404  color.w = color.w + value.w;
405  }
406  }
407  }

542  end[e].y = isaac_inverse_d[ 1] * end_p[e].x + isaac_inverse_d[ 5] * end_p[e].y + isaac_inverse_d[ 9] * end_p[e].z + isaac_inverse_d[13] * end_p[e].w;
543  end[e].z = isaac_inverse_d[ 2] * end_p[e].x + isaac_inverse_d[ 6] * end_p[e].y + isaac_inverse_d[10] * end_p[e].z + isaac_inverse_d[14] * end_p[e].w;
544  isaac_float max_size = isaac_size_d[0].max_global_size_scaled / 2.0f;
545 
546  //scale to globale grid size
547  start[e] = start[e] * max_size;
548  end[e] = end[e] * max_size;
549 
550  for (isaac_int i = 0; i < input_clipping.count; i++)
551  {
552  clipping[e].elem[i].position = input_clipping.elem[i].position * max_size;
553  clipping[e].elem[i].normal = input_clipping.elem[i].normal;
554  }
555 
556  //move to local (scaled) grid
557  move[e].x = isaac_int(isaac_size_d[0].global_size_scaled.value.x) / isaac_int(2) - isaac_int(isaac_size_d[0].position_scaled.value.x);
558  move[e].y = isaac_int(isaac_size_d[0].global_size_scaled.value.y) / isaac_int(2) - isaac_int(isaac_size_d[0].position_scaled.value.y);
559  move[e].z = isaac_int(isaac_size_d[0].global_size_scaled.value.z) / isaac_int(2) - isaac_int(isaac_size_d[0].position_scaled.value.z);
560 
561  move_f[e].x = isaac_float(move[e].x);
562  move_f[e].y = isaac_float(move[e].y);
563  move_f[e].z = isaac_float(move[e].z);
564 
565  start[e] = start[e] + move_f[e];
566  end[e] = end[e] + move_f[e];
567  for (isaac_int i = 0; i < input_clipping.count; i++)
568  clipping[e].elem[i].position = clipping[e].elem[i].position + move_f[e];
569 
570  vec[e] = end[e] - start[e];
571  l_scaled[e] = sqrt( vec[e].x * vec[e].x + vec[e].y * vec[e].y + vec[e].z * vec[e].z );
572 
573  start[e].x = start[e].x / scale.x;
574  start[e].y = start[e].y / scale.y;
575  start[e].z = start[e].z / scale.z;
576  end[e].x = end[e].x / scale.x;
577  end[e].y = end[e].y / scale.y;
578  end[e].z = end[e].z / scale.z;
579  for (isaac_int i = 0; i < input_clipping.count; i++)
580  {
581  clipping[e].elem[i].position.x = clipping[e].elem[i].position.x / scale.x;
582  clipping[e].elem[i].position.y = clipping[e].elem[i].position.y / scale.y;
583  clipping[e].elem[i].position.z = clipping[e].elem[i].position.z / scale.z;
584  }
585 
586  vec[e] = end[e] - start[e];
587  l[e] = sqrt( vec[e].x * vec[e].x + vec[e].y * vec[e].y + vec[e].z * vec[e].z );
588 
589  step_vec[e] = vec[e] / l[e] * step;
590  count_start[e] = - start[e] / step_vec[e];
591  local_size_f[e].x = isaac_float(isaac_size_d[0].local_size.value.x);
592  local_size_f[e].y = isaac_float(isaac_size_d[0].local_size.value.y);
593  local_size_f[e].z = isaac_float(isaac_size_d[0].local_size.value.z);
594

596 
597  //count_start shall have the smaller values
598  ISAAC_SWITCH_IF_SMALLER( count_end[e].x, count_start[e].x )
599  ISAAC_SWITCH_IF_SMALLER( count_end[e].y, count_start[e].y )
600  ISAAC_SWITCH_IF_SMALLER( count_end[e].z, count_start[e].z )
601 
602  //calc intersection of all three super planes and save in [count_start.x ; count_end.x]
603  count_start[e].x = ISAAC_MAX( ISAAC_MAX( count_start[e].x, count_start[e].y ), count_start[e].z );
604  count_end[e].x = ISAAC_MIN( ISAAC_MIN( count_end[e].x, count_end[e].y ), count_end[e].z );
605  if ( count_start[e].x > count_end[e].x)
606  {
607  if (!finish[e])
608  ISAAC_SET_COLOR( pixels[pixel[e].x + pixel[e].y * framebuffer_size.x], color[e] )
609  finish[e] = true;
610  }
611  }
612  ISAAC_ELEM_ALL_TRUE_RETURN( finish )

615  isaac_int last[ISAAC_VECTOR_ELEM];
616  isaac_float3 pos[ISAAC_VECTOR_ELEM];
617  isaac_int3 coord[ISAAC_VECTOR_ELEM];
618  isaac_float d[ISAAC_VECTOR_ELEM];
619  isaac_float intersection_step[ISAAC_VECTOR_ELEM];
620 
621  ISAAC_ELEM_ITERATE(e)
622  {
623  first[e] = isaac_int( floor(count_start[e].x) );
624  last[e] = isaac_int( ceil(count_end[e].x) );
625 
626  //Moving last and first until their points are valid
627  pos[e] = start[e] + step_vec[e] * isaac_float(last[e]);
628  coord[e].x = isaac_int(floor(pos[e].x));
629  coord[e].y = isaac_int(floor(pos[e].y));
630  coord[e].z = isaac_int(floor(pos[e].z));
631  while ( (ISAAC_FOR_EACH_DIM_TWICE(3, coord[e], >= isaac_size_d[0].local_size.value, || )
632  ISAAC_FOR_EACH_DIM (3, coord[e], < 0 || ) 0 ) && first[e] <= last[e])
633  {
634  last[e]--;
635  pos[e] = start[e] + step_vec[e] * isaac_float(last[e]);
636  coord[e].x = isaac_int(floor(pos[e].x));
637  coord[e].y = isaac_int(floor(pos[e].y));
638  coord[e].z = isaac_int(floor(pos[e].z));
639  }
640  pos[e] = start[e] + step_vec[e] * isaac_float(first[e]);
641  coord[e].x = isaac_int(floor(pos[e].x));
642  coord[e].y = isaac_int(floor(pos[e].y));
643  coord[e].z = isaac_int(floor(pos[e].z));
644  while ( (ISAAC_FOR_EACH_DIM_TWICE(3, coord[e], >= isaac_size_d[0].local_size.value, || )
645  ISAAC_FOR_EACH_DIM (3, coord[e], < 0 || ) 0 ) && first[e] <= last[e])
646  {
647  first[e]++;
648  pos[e] = start[e] + step_vec[e] * isaac_float(first[e]);
649  coord[e].x = isaac_int(floor(pos[e].x));
650  coord[e].y = isaac_int(floor(pos[e].y));
651  coord[e].z = isaac_int(floor(pos[e].z));
652  }
653 
654  //Extra clipping
655  for (isaac_int i = 0; i < input_clipping.count; i++)
656  {
657  d[e] = step_vec[e].x * clipping[e].elem[i].normal.x
658  + step_vec[e].y * clipping[e].elem[i].normal.y
659  + step_vec[e].z * clipping[e].elem[i].normal.z;
660  intersection_step[e] = ( clipping[e].elem[i].position.x * clipping[e].elem[i].normal.x
661  + clipping[e].elem[i].position.y * clipping[e].elem[i].normal.y
662  + clipping[e].elem[i].position.z * clipping[e].elem[i].normal.z
663  - start[e].x * clipping[e].elem[i].normal.x
664  - start[e].y * clipping[e].elem[i].normal.y
665  - start[e].z * clipping[e].elem[i].normal.z ) / d[e];
666  if (d[e] > 0)
667  {
668  if ( last[e] < intersection_step[e] )
669  {
670  if (!finish[e])
671  ISAAC_SET_COLOR( pixels[pixel[e].x + pixel[e].y * framebuffer_size.x], color[e] )
672  finish[e] = true;
673  }
674  if ( first[e] < intersection_step[e] )
675  first[e] = ceil( intersection_step[e] );
676  }
677  else
678  {
679  if ( first[e] > intersection_step[e] )
680  {
681  if (!finish[e])
682  ISAAC_SET_COLOR( pixels[pixel[e].x + pixel[e].y * framebuffer_size.x], color[e] )
683  finish[e] = true;
684  }
685  if ( last[e] > intersection_step[e] )
686  last[e] = floor( intersection_step[e] );
687  }
688  }
689  }

707  for (isaac_int i = first[e]; i <= last[e]; i++)

715  isaac_for_each_with_mpl_params

738  if (TIsoSurface)
739  {
740  if (result[e])
741  {
742  color[e] = value[e];
743  break;
744  }
745  }
746  else
747  {
748  oma[e] = isaac_float(1) - color[e].w;
749  value[e] = value[e] * factor[e];
750  color_add[e].x = oma[e] * value[e].x; // * value.w does merge_source_iterator
751  color_add[e].y = oma[e] * value[e].y; // * value.w does merge_source_iterator
752  color_add[e].z = oma[e] * value[e].z; // * value.w does merge_source_iterator
753  color_add[e].w = oma[e] * value[e].w;
754  color[e] = color[e] + color_add[e];
755  if (color[e].w > isaac_float(0.99))
756  break;
757  }
758  }
759  #if ISAAC_SHOWBORDER == 1

763  color_add[e].x = 0;
764  color_add[e].y = 0;

766  color_add[e].w = oma[e] * factor[e] * isaac_float(10);

770  #endif
